2. Create `src/components/session-status-indicator.tsx`:

```typescript
import { cn } from "@/lib/utils"

type SessionStatus = "streaming" | "unread" | "error" | "idle"

function SessionStatusIndicator({
  status,
}: {
  readonly status: SessionStatus
}) {
  if (status === "idle") return null

  return (
    <span
      className={cn(
        "size-1.5 shrink-0 rounded-full",
        status === "streaming" && "animate-pulse bg-sidebar-primary",
        status === "unread" && "bg-sidebar-primary",
        status === "error" && "bg-destructive",
      )}
    />
  )
}

export { SessionStatusIndicator }
export type { SessionStatus }
```

Per UI-SPEC:
- 6px diameter = `size-1.5` (Tailwind 1.5 * 4 = 6px)
- Streaming: `bg-sidebar-primary` + `animate-pulse`
- Unread: `bg-sidebar-primary` solid (no animation)
- Error: `bg-destructive` solid
- Idle: returns null (no indicator)
- `shrink-0` prevents flex compression

Component structure per UI-SPEC layout contract:

```tsx
import { HugeiconsIcon } from "@hugeicons/react"
import { SomeChevronIcon } from "@hugeicons/core-free-icons" // Find correct icon name
import {
  Sidebar,
  SidebarContent,
  SidebarGroup,
  SidebarGroupContent,
  SidebarGroupLabel,
  SidebarHeader,
  SidebarMenu,
  SidebarMenuButton,
  SidebarMenuItem,
  SidebarTrigger,
} from "@/components/ui/sidebar"
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "@/components/ui/collapsible"
import { SessionStatusIndicator } from "@/components/session-status-indicator"
import { mockProjects } from "@/atoms/sidebar"

const ACTIVE_TAB_ID = "tab-1"

function AppSidebar() {
  return (
    <Sidebar side="left" collapsible="offcanvas">
      <SidebarHeader className="flex flex-row items-center justify-between">
        <span className="text-base font-semibold">Mao</span>
        <SidebarTrigger />
      </SidebarHeader>
      <SidebarContent>
        {mockProjects.map((project) => (
          <Collapsible key={project.name} defaultOpen className="group/collapsible">
            <SidebarGroup>
              <SidebarGroupLabel
                render={<CollapsibleTrigger />}
                className="font-normal"
              >
                <HugeiconsIcon
                  icon={SomeChevronIcon}
                  className="transition-transform group-data-[state=closed]/collapsible:-rotate-90"
                  data-icon
                />
                {project.name}
              </SidebarGroupLabel>
              <CollapsibleContent>
                <SidebarGroupContent>
                  <SidebarMenu>
                    {project.sessions.map((session) => (
                      <SidebarMenuItem key={session.id}>
                        <SidebarMenuButton
                          isActive={session.id === ACTIVE_TAB_ID}
                        >
                          <span className="truncate">{session.branchLabel}</span>
                          <SessionStatusIndicator status={session.status} />
                        </SidebarMenuButton>
                      </SidebarMenuItem>
                    ))}
                  </SidebarMenu>
                </SidebarGroupContent>
              </CollapsibleContent>
            </SidebarGroup>
          </Collapsible>
        ))}
      </SidebarContent>
    </Sidebar>
  )
}

export { AppSidebar }
```

Key styling decisions per UI-SPEC:
- `SidebarGroupLabel` gets `className="font-normal"` to override shadcn default `font-medium` (stays within 2-weight budget: 400 + 600)
- Sidebar header: `text-base font-semibold` for "Mao" app name (heading typography per spec)
- Header layout: `flex flex-row items-center justify-between` for name + trigger
- Session branch labels use `truncate` for text overflow
- Chevron icon rotates via `group-data-[state=closed]/collapsible:-rotate-90 transition-transform` (Collapsible sets data-state on root). The `className="group/collapsible"` on `Collapsible` enables this selector.
- No `data-icon` sizing needed because icon is inside `SidebarGroupLabel` (not a Button)
