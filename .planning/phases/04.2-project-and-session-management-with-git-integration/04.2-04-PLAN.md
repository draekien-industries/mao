    ```typescript
    import { Atom } from "@effect-atom/atom-react";
    import { Effect } from "effect";
    import type { Project } from "@/services/database/project-store/schemas";
    import type { Tab } from "@/services/database/tab-store/schemas";
    import { RendererRpcClient } from "@/services/claude-rpc/client";
    import { appRuntime } from "./runtime";

    export interface ProjectWithSessions {
      readonly project: Project;
      readonly sessions: ReadonlyArray<Tab>;
    }

    export const projectsAtom = Atom.make<ReadonlyArray<ProjectWithSessions>>([]).pipe(
      Atom.keepAlive,
    );

    export const activeTabIdAtom = Atom.make<number | null>(null).pipe(Atom.keepAlive);

    export const loadProjectsAtom = appRuntime.fn(
      (_: void, ctx: Atom.FnContext) =>
        Effect.gen(function* () {
          const client = yield* RendererRpcClient;
          const projects = yield* client.listProjects({});
          const tabs = yield* client.listTabs({});

          const grouped: ProjectWithSessions[] = projects.map((project) => ({
            project,
            sessions: tabs.filter((tab) => tab.project_id === project.id),
          }));

          ctx.set(projectsAtom, grouped);
        }),
    );
    ```

    import { useEffect } from "react";
    import { ArrowDown01Icon } from "@hugeicons/core-free-icons";
    import { HugeiconsIcon } from "@hugeicons/react";
    import { tabStatusAtom } from "@/atoms/chat";
    import {
      activeTabIdAtom,
      loadProjectsAtom,
      projectsAtom,
    } from "@/atoms/sidebar";
    import type { Tab } from "@/services/database/tab-store/schemas";
    import { SessionStatusIndicator } from "@/components/session-status-indicator";
    import { Button } from "@/components/ui/button";
    // ... sidebar UI imports (unchanged)

    function SessionEntry({
      isActive,
      session,
    }: {
      readonly isActive: boolean;
      readonly session: Tab;
    }) {
      const liveStatus = useAtomValue(tabStatusAtom(String(session.id)));

      return (
        <SidebarMenuItem>
          <SidebarMenuButton isActive={isActive}>
            <span className="truncate">
              {session.git_branch ?? session.display_label ?? "untitled"}
            </span>
            <SessionStatusIndicator status={liveStatus} />
          </SidebarMenuButton>
        </SidebarMenuItem>
      );
    }

    function AppSidebar() {
      const projects = useAtomValue(projectsAtom);
      const activeTabId = useAtomValue(activeTabIdAtom);
      const loadProjects = loadProjectsAtom;

      useEffect(() => {
        loadProjects();
      }, []);

      return (
        <Sidebar collapsible="offcanvas" side="left">
          <SidebarHeader>
            <span className="text-base font-semibold">Mao</span>
          </SidebarHeader>
          <SidebarContent>
            {projects.length === 0 ? (

                <p className="text-sm font-semibold">No projects yet</p>
                <p className="text-xs text-muted-foreground">
                  Register a project folder to get started.
                </p>
                <Button variant="default" size="sm">
                  Register Project

              </div>

                  defaultOpen

                      className="font-normal"

                      <HugeiconsIcon
                        className="transition-transform group-data-[closed]/collapsible:-rotate-90"
                        icon={ArrowDown01Icon}
                      />
                      {project.name}
                    </SidebarGroupLabel>
                    <CollapsibleContent>
                      <SidebarGroupContent>
                        <SidebarMenu>
                          {sessions.map((session) => (
                            <SessionEntry
                              isActive={session.id === activeTabId}
                              key={session.id}
                              session={session}
                            />
                          ))}
                        </SidebarMenu>
                      </SidebarGroupContent>
                    </CollapsibleContent>
                  </SidebarGroup>
                </Collapsible>
              ))
            )}
          </SidebarContent>
        </Sidebar>
      );
    }
