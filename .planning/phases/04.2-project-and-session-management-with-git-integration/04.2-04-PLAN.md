    ```typescript
    import { Context } from "effect";

    // Tag for accessing the full typed RPC client in the renderer
    type MergedRpcClient = RpcClient.RpcClient<typeof MergedRpcGroup>;

    export class RendererRpcClient extends Context.Tag("RendererRpcClient")<
      RendererRpcClient,
      MergedRpcClient
    >() {}

    export const RendererRpcClientLayer = Layer.scoped(
      RendererRpcClient,
      RpcClient.make(MergedRpcGroup),
    ).pipe(Layer.provide(ElectronClientProtocol));
    ```

    4. **loadProjectsEffect** -- shared helper for reloading projects from DB:
    ```typescript
    const loadProjectsEffect = (ctx: Atom.FnContext) =>
      Effect.gen(function* () {
        const client = yield* RendererRpcClient;
        const projects = yield* client.listProjects({});
        const tabs = yield* client.listTabs({});
        const grouped: ProjectWithSessions[] = projects.map((project) => ({
          project,
          sessions: tabs.filter((tab) => tab.project_id === project.id),
        }));
        ctx.set(projectsAtom, grouped);
      });
    ```
